player_controller.py 
from InfEngine import InfComponent, vec3f
from InfEngine.components import serialized_field

class PlayerController(InfComponent):
    speed: float = serialized_field(
        default=5.0,
        range=(0, 100),
        tooltip="Movement speed"
    )

    def start(self):
        Debug.log(f"Speed: {self.speed}")

    def update(self, dt: float):
        pos = self.transform.position
        self.transform.position = vec3f(
            pos.x + self.speed * dt,
            pos.y, pos.z
        )

Why Python as Game Engine Frontend?

🤖

AI Integration is Native

Import PyTorch, TensorFlow, OpenCV directly in your game scripts. No plugin layer. No bridging. Just import torch.

✨

Vibe Coding Ready

LLMs generate Python more accurately than any other language. Your game logic is naturally LLM-friendly.

📊

NumPy Render Readback

Read render pass pixels directly as NumPy arrays. Perfect for real-time CV, RL, and GPU-to-CPU pipelines.

🎮

Unity-Style API

Familiar Component-GameObject architecture with lifecycle methods. URP-inspired Scriptable Render Pipeline in Python.

🔥

Vulkan Powered

C++17 backend with Vulkan for maximum GPU performance. Python defines what to render, C++ handles how.

🔓

Open Source

MIT licensed. No royalties, no restrictions. Full access to engine source code.

  • Directional shadow mapping (PCF)
  • Post-processing pipeline

Render Architecture

  • Transform hierarchy (parent-child)

rotator.py 
from InfEngine import InfComponent, vec3f
from InfEngine.components import serialized_field, require_component
from InfEngine.debug import Debug

@require_component("MeshRenderer")
class Rotator(InfComponent):
    """Rotates the GameObject continuously."""

    speed: float = serialized_field(default=45.0, tooltip="Degrees per second")
    axis: str = serialized_field(default="y")

    def start(self):
        Debug.log(f"Rotator started: {self.speed}°/s on {self.axis}")

    def update(self, delta_time: float):
        rot = self.transform.euler_angles
        delta = self.speed * delta_time
        if self.axis == "y":
            rot.y += delta
        elif self.axis == "x":
            rot.x += delta
        self.transform.euler_angles = rot
my_pipeline.py 
from InfEngine.rendering import RenderPipeline
from InfEngine.rendergraph import RenderGraph, Format

class MyPipeline(RenderPipeline):
    def render(self, context, cameras):
        for camera in cameras:
            context.setup_camera_properties(camera)
            culling = context.cull(camera)  # frustum culling

            graph = RenderGraph("ForwardPipeline")
            color = graph.backbuffer()
            depth = graph.create_texture("Depth",
                format=Format.D32_SFLOAT)

            opaque = graph.add_pass("Opaque", priority=0)
            opaque.write_color(color)
            opaque.write_depth(depth)
            opaque.set_clear(
                color=(0.1, 0.1, 0.15, 1.0),
                depth=1.0
            )
            opaque.draw_renderers(
                queue_range=(0, 2500),
                sort_mode="front_to_back"
            )

            graph.set_output(color)
            context.apply_graph(graph.build())
            context.submit_culling(culling)
